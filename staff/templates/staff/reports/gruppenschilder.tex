%% Gibt ein Gruppenschild aus
% 1: Name
% 2: Bild
\newcommand{\Gruppe}[2]{
	\begin{minipage}{\textwidth}
		\begin{center}
		\fontsize{100pt}{0}\selectfont
		\vspace{-1cm}\includegraphics[width=0.9\textwidth]{#2}
		\vspace{1cm}
		\end{center}
		\tcboxfit[width=\textwidth,height=0.3\textheight, halign=center]{
		\fontsize{100pt}{0}\selectfont
		\centering
			\sf{#1} \\}
	\end{minipage}
	\newpage
}
